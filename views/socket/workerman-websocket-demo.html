<?php

date_default_timezone_set("PRC");

use Workerman\Connection\TcpConnection;
use Workerman\Worker;
use Workerman\Timer;

require_once __DIR__.'/workerman/vendor/autoload.php';

use Workerman\Crontab\Crontab;

function curl($url, $post = [], $type = 1, $header = [])
{
    $ch = curl_init();
    $_header = [];
    if ($header) {
        foreach ($header as $key => $val) {
            $_header[] = "{$key}:{$val}";
        }
    }
    if ($_header) {
        curl_setopt($ch, CURLOPT_HTTPHEADER, $_header);               // еҠ е…Ҙheader
    }
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, 0);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
    curl_setopt($ch, CURLOPT_HEADER, FALSE);
    if (strpos($url, 'https') !== FALSE) {
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    }
    if (!empty($post)) {
        curl_setopt($ch, CURLOPT_POST, TRUE);
        if ($type == 1) {
            curl_setopt($ch, CURLOPT_POSTFIELDS, $post);
        } else {
            curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query($post));
        }
    }
    curl_setopt($ch, CURLOPT_TIMEOUT, 10);
    $output = curl_exec($ch);
    $status = curl_getinfo($ch, CURLINFO_HTTP_CODE);//иҺ·еҸ–httpиҝ”еӣһеҖј,жңҖеҗҺдёҖдёӘж”¶еҲ°зҡ„HTTPд»Јз Ғ
    if ($output == false) {
        echo "status: ".$status.PHP_EOL;
        echo "curl error no: ".curl_errno($ch).PHP_EOL;
        echo "curl_error".curl_errno($ch).PHP_EOL;
    }
    curl_close($ch);
    return $output;
}

$ws_worker = new Worker('websocket://0.0.0.0:8080');
$ws_worker->count = 1;

// иҝһжҺҘе»әз«Ӣж—¶з»ҷеҜ№еә”иҝһжҺҘи®ҫзҪ®е®ҡж—¶еҷЁ
$ws_worker->onConnect = function (TcpConnection $connection) {
    // жҜҸ10з§’жү§иЎҢдёҖж¬Ў
    $connection->send(json_encode(['status' => 1, 'msg' => 'иҝһжҺҘжҲҗеҠҹ']));
};

$ws_worker->onMessage = function (TcpConnection $connection, $data) {
    $recv_data = json_decode($data, true);
    $connection->send(json_encode(['status' => 1, 'msg' => $recv_data['switch'] == 1 ? 'ејҖеҗҜеҸ‘йҖҒж¶ҲжҒҜ': 'е…ій—ӯеҸ‘йҖҒж¶ҲжҒҜ']));
    if (isset($recv_data['switch']) && $recv_data['switch'] == 2) {
        // еҲ йҷӨе®ҡж—¶еҷЁ
        Timer::del($connection->timer_id);
    } else {
        $time_interval = 1;
        // з»ҷconnectionеҜ№иұЎдёҙж—¶ж·»еҠ дёҖдёӘtimer_idеұһжҖ§дҝқеӯҳе®ҡж—¶еҷЁid
        $connection->timer_id = Timer::add(
            $time_interval,
            function ($connection) {
                $res = curl("http://127.0.0.1/Auto/autoSend");
                $connection->send(json_encode(['status' => 1, 'msg' => $res]));
            },
            array($connection)
        );
    }
};

// иҝһжҺҘе…ій—ӯж—¶пјҢеҲ йҷӨеҜ№еә”иҝһжҺҘзҡ„е®ҡж—¶еҷЁ
$ws_worker->onClose = function (TcpConnection $connection) {
    // еҲ йҷӨе®ҡж—¶еҷЁ
    Timer::del($connection->timer_id);
};

// иҝҗиЎҢworker
Worker::runAll();
