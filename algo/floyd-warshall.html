дҪ жү“ејҖй«ҳеҫ·ең°еӣҫпјҢиҫ“е…Ҙ"д»Һе®¶еҲ°е…¬еҸё"пјҢе®ғз§’еӣһдёҖжқЎжңҖдјҳи·ҜзәҝгҖӮдҪҶеҰӮжһңиҖҒжқҝзӘҒз„¶й—®дҪ пјҡ"е’ұд»¬е…¬еҸёеҲ°жүҖжңүеҲҶйғЁд№Ӣй—ҙпјҢдёӨдёӨд№Ӣй—ҙзҡ„жңҖзҹӯи·қзҰ»жҳҜеӨҡе°‘пјҹ"дҪ жҖ»дёҚиғҪдёҖдёӘдёӘеҺ»жҹҘеҗ§пјҹеҰӮжһңжңү 100 дёӘеҲҶйғЁпјҢе°ұиҰҒжҹҘ 100 Г— 99 = 9900 ж¬ЎгҖӮ

иҝҷе°ұжҳҜе…ЁжәҗжңҖзҹӯи·Ҝеҫ„й—®йўҳпјҡз»ҷе®ҡдёҖдёӘеёҰжқғеӣҫпјҢжұӮеҮәжҜҸдёҖеҜ№йЎ¶зӮ№д№Ӣй—ҙзҡ„жңҖзҹӯи·Ҝеҫ„гҖӮ

д№ӢеүҚжҲ‘д»¬иҒҠиҝҮ Dijkstra з®—жі•пјҢе®ғж“…й•ҝи§ЈеҶіеҚ•жәҗжңҖзҹӯи·Ҝеҫ„вҖ”вҖ”д»ҺдёҖдёӘзӮ№еҮәеҸ‘еҲ°е…¶д»–жүҖжңүзӮ№зҡ„жңҖзҹӯи·қзҰ»гҖӮи·‘ n ж¬Ў Dijkstra еҪ“з„¶иғҪжҗһе®ҡе…ЁжәҗжңҖзҹӯи·Ҝеҫ„пјҢдҪҶ Floyd-Warshall з»ҷдәҶжҲ‘д»¬дёҖдёӘжӣҙдјҳйӣ…зҡ„йҖүжӢ©пјҡж ёеҝғд»Јз ҒеҸӘжңү 5 иЎҢпјҢиҖҢдё”иҝҳиғҪеӨ„зҗҶиҙҹжқғиҫ№ рҹҺү

1. ж ёеҝғзӣҙи§ү вҖӢ

  • дҪ еҸҜд»ҘзӣҙиҫҫпјҡA вҶ’ DпјҢи·қзҰ» 10

4. иҙҹжқғиҫ№е’ҢиҙҹзҺҜжЈҖжөӢ вҖӢ

Floyd-Warshall жңүдёҖдёӘ Dijkstra еҒҡдёҚеҲ°зҡ„дәӢпјҡеӨ„зҗҶиҙҹжқғиҫ№гҖӮ

Dijkstra зҡ„иҙӘеҝғзӯ–з•Ҙдҫқиө–"е·ІзЎ®е®ҡзҡ„жңҖзҹӯи·Ҝеҫ„дёҚдјҡеҶҚеҸҳ"пјҢдҪҶиҙҹжқғиҫ№дјҡжү“з ҙиҝҷдёӘеҒҮи®ҫгҖӮиҖҢ Floyd-Warshall з”Ёзҡ„жҳҜеҠЁжҖҒи§„еҲ’пјҢдёҚдҫқиө–иҝҷдёӘжҖ§иҙЁпјҢжүҖд»ҘеӨ©з„¶ж”ҜжҢҒиҙҹжқғиҫ№гҖӮ

дҪҶиҰҒжіЁж„ҸпјҡеҰӮжһңеӣҫдёӯеӯҳеңЁиҙҹзҺҜпјҲжҖ»жқғеҖјдёәиҙҹзҡ„зҺҜпјүпјҢжңҖзҹӯи·Ҝеҫ„е°ұжҳҜиҙҹж— з©·вҖ”вҖ”дҪ еҸҜд»Ҙж— йҷҗз»•еңҲжҠҠи·қзҰ»жӢүеҲ°иҙҹж— з©·гҖӮFloyd-Warshall иғҪеё®дҪ жЈҖжөӢиҝҷз§Қжғ…еҶөпјҡи·‘е®Ңд№ӢеҗҺпјҢеҰӮжһңжҹҗдёӘйЎ¶зӮ№еҲ°иҮӘиә«зҡ„и·қзҰ» dist[i][i] < 0пјҢиҜҙжҳҺеӯҳеңЁиҙҹзҺҜгҖӮ

иҙҹзҺҜзӨәдҫӢпјҡ

    A вҶ’ BпјҲжқғ -2пјү
    B вҶ’ CпјҲжқғ -3пјү
    C вҶ’ AпјҲжқғ 1пјү

    жҜҸиө°дёҖеңҲи·қзҰ»е°ұе°‘ 4пјҢж— йҷҗеҫӘзҺҜ вҶ’ жңҖзҹӯи·Ҝеҫ„ = -вҲһ

 * ж ёеҝғжҖқи·ҜпјҡдёүйҮҚеҫӘзҺҜпјҢйҖҗдёӘе°қиҜ•"з»ҸиҝҮжҜҸдёӘйЎ¶зӮ№дҪңдёәдёӯй—ҙзӮ№"пјҢ
 * зңӢиғҪдёҚиғҪжүҫеҲ°жӣҙзҹӯзҡ„и·Ҝеҫ„гҖӮ
 *
 * ж—¶й—ҙеӨҚжқӮеәҰ O(nВі)пјҢз©әй—ҙеӨҚжқӮеәҰ O(nВІ)
 * йҖӮеҗҲйЎ¶зӮ№ж•° < 1000 зҡ„зЁ еҜҶеӣҫ
 */

const INF = Number.MAX_SAFE_INTEGER;

function floydWarshall(graph: number[][]): number[][] {
  const n = graph.length;

  // еҲқе§ӢеҢ–и·қзҰ»зҹ©йҳөпјҡжӢ·иҙқдёҖд»ҪеҺҹеӣҫзҡ„йӮ»жҺҘзҹ©йҳө
  // дёәд»Җд№ҲжӢ·иҙқпјҡжҲ‘д»¬дјҡеңЁ dist дёҠеҺҹең°дҝ®ж”№пјҢдёҚжғіж”№еҸҳеҺҹеӣҫ
  const dist: number[][] = graph.map((row) => [...row]);

  // ж ёеҝғд»Јз Ғпјҡзңҹзҡ„еҸӘжңү 5 иЎҢ
  // k = дёӯй—ҙзӮ№пјҲеҝ…йЎ»еңЁжңҖеӨ–еұӮпјҒиҝҷжҳҜжӯЈзЎ®жҖ§зҡ„е…ій”®пјү
  for (let k = 0; k < n; k++) {
    for (let i = 0; i < n; i++) {
      for (let j = 0; j < n; j++) {
        // еҰӮжһң iвҶ’k е’Ң kвҶ’j йғҪеҸҜиҫҫпјҢдё”з»ҸиҝҮ k жӣҙзҹӯ
        if (dist[i][k] !== INF && dist[k][j] !== INF) {
          dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
        }
      }
    }
  }

  return dist;
}

/**
 * еҲӨж–ӯеӣҫдёӯжҳҜеҗҰеӯҳеңЁиҙҹзҺҜ
 * и·‘е®Ң Floyd-Warshall еҗҺпјҢеҰӮжһң dist[i][i] < 0пјҢиҜҙжҳҺеӯҳеңЁиҙҹзҺҜ
 */
function hasNegativeCycle(dist: number[][]): boolean {
  for (let i = 0; i < dist.length; i++) {
    if (dist[i][i] < 0) return true;
  }
  return false;
}

/**
 * йҮҚе»әжңҖзҹӯи·Ҝеҫ„пјҲдёҚеҸӘжҳҜи·қзҰ»пјҢиҝҳиғҪиҫ“еҮәе…·дҪ“и·Ҝеҫ„пјү
 *
 * жҖқи·ҜпјҡйўқеӨ–з»ҙжҠӨдёҖдёӘ next зҹ©йҳөпјҢnext[i][j] и®°еҪ• iвҶ’j жңҖзҹӯи·ҜдёҠзҡ„з¬¬дёҖдёӘдёӯиҪ¬зӮ№
 * жҜҸж¬Ўжӣҙж–°и·қзҰ»ж—¶пјҢеҗҢжӯҘжӣҙж–° next
 */
function floydWarshallWithPath(graph: number[][]): {
  dist: number[][];
  next: number[][];
} {
  const n = graph.length;
  const dist: number[][] = graph.map((row) => [...row]);

  // next[i][j] = д»Һ i еҮәеҸ‘еҲ° jпјҢз¬¬дёҖдёӘиҰҒз»ҸиҝҮзҡ„йЎ¶зӮ№
  // еҲқе§ӢеҢ–пјҡеҰӮжһң iвҶ’j зӣҙиҝһпјҢnext[i][j] = jпјӣеҗҰеҲҷ -1
  const next: number[][] = Array.from({ length: n }, (_, i) =>
    Array.from({ length: n }, (_, j) => (graph[i][j] !== INF ? j : -1)),
  );

  for (let k = 0; k < n; k++) {
    for (let i = 0; i < n; i++) {
      for (let j = 0; j < n; j++) {
        if (dist[i][k] !== INF && dist[k][j] !== INF) {
          if (dist[i][k] + dist[k][j] < dist[i][j]) {
            dist[i][j] = dist[i][k] + dist[k][j];
            next[i][j] = next[i][k]; // и·Ҝеҫ„з»ҸиҝҮ kпјҢжӣҙж–°дёӯиҪ¬зӮ№
          }
        }
      }
    }
  }

  return { dist, next };
}

/**
 * жҹҘиҜўд»Һ start еҲ° end зҡ„е…·дҪ“жңҖзҹӯи·Ҝеҫ„
 */
function getPath(
  next: number[][],
  start: number,
  end: number,
): number[] {
  if (next[start][end] === -1) return []; // дёҚеҸҜиҫҫ
  const path = [start];
  let current = start;
  while (current !== end) {
    current = next[current][end];
    path.push(current);
  }
  return path;
}

// ===== дҪҝз”ЁзӨәдҫӢ =====

//       в‘  --3-- в‘Ў
//       |  \     |
//      10   5    2
//       |     \  |
//       в‘ў --6-- в‘Ј

const graph = [
  //  в‘          в‘Ў         в‘ў         в‘Ј
  [0, 3, 10, 5],       // в‘  еҲ°еҗ„зӮ№
  [INF, 0, INF, 2],    // в‘Ў еҲ°еҗ„зӮ№
  [INF, INF, 0, 6],    // в‘ў еҲ°еҗ„зӮ№
  [INF, INF, INF, 0],  // в‘Ј еҲ°еҗ„зӮ№
];

const result = floydWarshall(graph);
console.log("жңҖзҹӯи·қзҰ»зҹ©йҳөпјҡ");
result.forEach((row, i) => {
  console.log(`д»Һв‘ в‘Ўв‘ўв‘Ј[i] вҶ’ `, row.map((d) => (d === INF ? "вҲһ" : d)));

"""
Floyd-Warshall е…ЁжәҗжңҖзҹӯи·Ҝеҫ„з®—жі• вҖ”вҖ” Python е®һзҺ°

ж ёеҝғжҖқи·ҜпјҡдёүйҮҚеҫӘзҺҜпјҢйҖҗдёӘе°қиҜ•з»ҸиҝҮжҜҸдёӘйЎ¶зӮ№дҪңдёәдёӯиҪ¬зӮ№
k еҝ…йЎ»еңЁжңҖеӨ–еұӮпјҒиҝҷжҳҜжӯЈзЎ®жҖ§зҡ„е…ій”®вҖ”вҖ”дҝқиҜҒз”Ё k еҒҡдёӯиҪ¬ж—¶пјҢ
iвҶ’k е’Ң kвҶ’j зҡ„жңҖзҹӯи·Ҝеҫ„е·Із»ҸзЎ®е®ҡгҖӮ
"""

INF = float("inf")


def floyd_warshall(graph: list[list[float]]) -> list[list[float]]:
    """жұӮжүҖжңүйЎ¶зӮ№еҜ№д№Ӣй—ҙзҡ„жңҖзҹӯи·қзҰ»"""
    n = len(graph)
    # жӢ·иҙқдёҖд»ҪпјҢдёҚеңЁеҺҹеӣҫдёҠж”№
    dist = [row[:] for row in graph]

    # ж ёеҝғд»Јз ҒпјҢзңҹзҡ„еҸӘжңү 5 иЎҢ
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] != INF and dist[k][j] != INF:
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])

    return dist


def has_negative_cycle(dist: list[list[float]]) -> bool:
    """жЈҖжҹҘжҳҜеҗҰеӯҳеңЁиҙҹзҺҜпјҡеҜ№и§’зәҝжңүиҙҹеҖје°ұиҜҙжҳҺеӯҳеңЁ"""
    return any(dist[i][i] < 0 for i in range(len(dist)))


def floyd_warshall_with_path(graph: list[list[float]]):
    """дёҚд»…з®—и·қзҰ»пјҢиҝҳиғҪйҮҚе»әе…·дҪ“и·Ҝеҫ„"""
    n = len(graph)
    dist = [row[:] for row in graph]

    # next[i][j] = д»Һ i еҲ° j жңҖзҹӯи·ҜдёҠзҡ„з¬¬дёҖдёӘдёӯиҪ¬зӮ№
    next_matrix = [
        [j if graph[i][j] != INF else -1 for j in range(n)]
        for i in range(n)
    ]

    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] != INF and dist[k][j] != INF:
                    if dist[i][k] + dist[k][j] < dist[i][j]:
                        dist[i][j] = dist[i][k] + dist[k][j]
                        next_matrix[i][j] = next_matrix[i][k]

    return dist, next_matrix


def get_path(next_matrix: list[list[int]], start: int, end: int) -> list[int]:
    """жҹҘиҜўд»Һ start еҲ° end зҡ„е…·дҪ“жңҖзҹӯи·Ҝеҫ„"""
    if next_matrix[start][end] == -1:
        return []  # дёҚеҸҜиҫҫ
    path = [start]
    current = start
    while current != end:
        current = next_matrix[current][end]
        path.append(current)
    return path


# ===== дҪҝз”ЁзӨәдҫӢ =====
if __name__ == "__main__":
    #       в‘  --3-- в‘Ў
    #       |  \     |
    #      10   5    2
    #       |     \  |
    #       в‘ў --6-- в‘Ј

    graph = [
        [0, 3, 10, 5],
        [INF, 0, INF, 2],
        [INF, INF, 0, 6],
        [INF, INF, INF, 0],
    ]

    result = floyd_warshall(graph)
    for row in result:
        print([d if d != INF else "вҲһ" for d in row])
    # [0, 3, 10, 5] вҖ”вҖ” д»Һв‘ еҲ°в‘ЈжңҖзҹӯи·қзҰ» 5

    dist, next_m = floyd_warshall_with_path(graph)
    path = get_path(next_m, 0, 3)
    print(f"в‘ вҶ’в‘Ј жңҖзҹӯи·Ҝеҫ„: {' вҶ’ '.join(str(p+1) for p in path)}")
