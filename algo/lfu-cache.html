LFU зј“еӯҳпјҲLeast Frequently Usedпјү вҖӢ

дёҠдёҖзҜҮжҲ‘д»¬иҒҠдәҶ LRU зј“еӯҳпјҢж·ҳжұ°зҡ„жҳҜгҖҢжңҖд№…жІЎз”ЁгҖҚзҡ„йӮЈдёӘгҖӮдҪҶдҪ жңүжІЎжңүжғіиҝҮдёҖдёӘй—®йўҳвҖ”вҖ”еҰӮжһңжҹҗдёӘж•°жҚ®иў«з–ҜзӢӮи®ҝй—®дәҶ 1000 ж¬ЎпјҢз„¶еҗҺе®үйқҷдәҶ 5 еҲҶй’ҹпјӣеҸҰдёҖдёӘж•°жҚ®еҸӘиў«и®ҝй—®дәҶ 1 ж¬ЎпјҢдҪҶе°ұеңЁеҲҡжүҚгҖӮLRU дјҡжҠҠйӮЈдёӘиў«и®ҝй—® 1000 ж¬Ўзҡ„ж•°жҚ®ж·ҳжұ°жҺүпјҢеӣ дёәе®ғжҳҜгҖҢжңҖиҝ‘жңҖе°‘дҪҝз”ЁгҖҚзҡ„гҖӮдҪҶзӣҙи§үдёҠпјҢиў«и®ҝй—® 1000 ж¬Ўзҡ„ж•°жҚ®жҳҺжҳҫжӣҙйҮҚиҰҒпјҢеҜ№еҗ§пјҹ

иҝҷе°ұжҳҜ LFUпјҲLeast Frequently UsedпјүиҰҒи§ЈеҶізҡ„й—®йўҳпјҡж·ҳжұ°дҪҝз”Ёйў‘зҺҮжңҖдҪҺзҡ„ж•°жҚ®гҖӮеҰӮжһңйў‘зҺҮзӣёеҗҢпјҢеҶҚж·ҳжұ°жңҖд№…жІЎз”Ёзҡ„йӮЈдёӘпјҲйҖҖеҢ–жҲҗ LRU зҡ„йҖ»иҫ‘пјүгҖӮ

еҗ¬иө·жқҘеҘҪеғҸжҜ” LRU йҡҫдёҖзӮ№пјҹжІЎй”ҷпјҢLFU жҳҜ LeetCode 460пјҢдёҖйҒ“ Hard йўҳгҖӮдҪҶеҲ«жҖ•пјҢд»ҠеӨ©жҲ‘д»¬жҠҠе®ғжӢҶзўҺдәҶи®І вңҢпёҸ

дёәд»Җд№ҲйңҖиҰҒ LFUпјҹ вҖӢ

е…ҲжқҘжҗһжё…жҘҡд»Җд№Ҳж—¶еҖҷиҜҘз”Ё LFUпјҢд»Җд№Ҳж—¶еҖҷиҜҘз”Ё LRUгҖӮ

еңәжҷҜжҺЁиҚҗзӯ–з•ҘеҺҹеӣ 
зҪ‘йЎөзј“еӯҳгҖҒAPI йҷҗжөҒLRUжңҖиҝ‘и®ҝй—®зҡ„еӨ§жҰӮзҺҮиҝҳдјҡи®ҝй—®пјҲж—¶й—ҙеұҖйғЁжҖ§пјү
зғӯй—Ёе•Ҷе“ҒгҖҒзғӯжҗңиҜҚзј“еӯҳLFUзңҹжӯЈзғӯй—Ёзҡ„ж•°жҚ®дёҚеә”иҜҘеӣ дёәгҖҢе®үйқҷдәҶдёҖдјҡе„ҝгҖҚе°ұиў«ж·ҳжұ°
ж•°жҚ®еә“жҹҘиҜўзј“еӯҳзңӢжғ…еҶөеҰӮжһңи®ҝй—®йў‘зҺҮеҲҶеёғдёҚеқҮеҢҖпјҢLFU жӣҙеҘҪ

дёҫдёӘе®һйҷ…зҡ„дҫӢеӯҗпјҡдҪ еңЁеҒҡдёҖдёӘз”өе•Ҷзі»з»ҹпјҢзј“еӯҳе•Ҷе“ҒиҜҰжғ…йЎөгҖӮiPhone зҡ„иҜҰжғ…йЎөжҜҸеӨ©иў«и®ҝй—® 100 дёҮж¬ЎпјҢжҹҗдёӘе°Ҹдј—иҖіжңәжҜҸеӨ©еҸӘиў«и®ҝй—® 50 ж¬ЎгҖӮеҰӮжһңз”Ё LRUпјҢеҪ“зј“еӯҳж»ЎдәҶгҖҒеҲҡеҘҪжңүдәәжҗңдәҶ 100 дёӘе°Ҹдј—е•Ҷе“ҒпјҢiPhone зҡ„зј“еӯҳе°ұиў«жҢӨеҮәеҺ»дәҶвҖ”вҖ”иҝҷжҳҫз„¶дёҚеҗҲзҗҶгҖӮз”Ё LFU зҡ„иҜқпјҢiPhone зҡ„йў‘зҺҮиҝңй«ҳдәҺе°Ҹдј—е•Ҷе“ҒпјҢж°ёиҝңдёҚдјҡиў«ж·ҳжұ°гҖӮ

еҺҹзҗҶжӢҶи§Ј вҖӢ

LFU зј“еӯҳиҰҒж”ҜжҢҒдёӨдёӘж“ҚдҪңпјҡ

  1. get(key)пјҡжҹҘжүҫ keyпјҢе‘ҪдёӯеҗҺиҜҘ key зҡ„йў‘зҺҮ +1
  2. put(key, value)пјҡжҸ’е…ҘжҲ–жӣҙж–° keyпјҢйў‘зҺҮеҲқе§ӢеҢ–дёә 1гҖӮеҰӮжһңзј“еӯҳж»ЎдәҶпјҢж·ҳжұ°йў‘зҺҮжңҖдҪҺзҡ„ keyпјҲйў‘зҺҮзӣёеҗҢеҲҷж·ҳжұ°жңҖд№…жІЎз”Ёзҡ„пјү

е…ій”®йҡҫзӮ№пјҡжүҖжңүж“ҚдҪңйғҪиҰҒ O(1)гҖӮиҝҷж„Ҹе‘ізқҖжҲ‘д»¬дёҚиғҪйҒҚеҺҶжүҖжңү key еҺ»жүҫйў‘зҺҮжңҖдҪҺзҡ„гҖӮ

ж•°жҚ®з»“жһ„и®ҫи®Ў вҖӢ

жҲ‘д»¬йңҖиҰҒдёүдёӘдёңиҘҝй…ҚеҗҲпјҡ

1. keyTable: Map<key, Node>
   - йҖҡиҝҮ key зӣҙжҺҘжүҫеҲ°иҠӮзӮ№пјҢO(1) жҹҘжүҫ

2. freqTable: Map<freq, DoublyLinkedList>
   - йҖҡиҝҮйў‘зҺҮжүҫеҲ°еҜ№еә”зҡ„еҸҢеҗ‘й“ҫиЎЁ
   - еҗҢдёҖйў‘зҺҮзҡ„жүҖжңү key жҢүи®ҝй—®ж—¶й—ҙжҺ’еәҸпјҲжңҖиҝ‘и®ҝй—®зҡ„еңЁеүҚпјү

3. minFreq: number
   - и®°еҪ•еҪ“еүҚжңҖе°Ҹйў‘зҺҮпјҢж·ҳжұ°ж—¶зӣҙжҺҘе®ҡдҪҚеҲ° freqTable[minFreq]

typescript
class DLinkedList {
  head: DLinkedNode; // е“Ёе…өеӨҙ
  tail: DLinkedNode; // е“Ёе…өе°ҫ
  size: number = 0;

  constructor() {
    this.head = new DLinkedNode(0, 0);
    this.tail = new DLinkedNode(0, 0);
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }

  // еңЁеӨҙйғЁжҸ’е…ҘпјҲжңҖиҝ‘и®ҝй—®зҡ„ж”ҫеүҚйқўпјү
  addFirst(node: DLinkedNode): void {
    node.prev = this.head;
    node.next = this.head.next;
    this.head.next!.prev = node;
    this.head.next = node;
    this.size++;
  }

  // еҲ йҷӨиҠӮзӮ№
  remove(node: DLinkedNode): void {
    node.prev!.next = node.next;
    node.next!.prev = node.prev;
    this.size--;
  }

  // еҲ йҷӨе№¶иҝ”еӣһе°ҫйғЁиҠӮзӮ№пјҲжңҖд№…жІЎз”Ёзҡ„пјү
  removeLast(): DLinkedNode | null {
    if (this.size === 0) return null;
    const last = this.tail.prev!;
    this.remove(last);
    return last;
  }
}

class LFUCache {
  private capacity: number;
  private minFreq: number = 0;
  private keyTable: Map<number, DLinkedNode> = new Map();
  private freqTable: Map<number, DLinkedList> = new Map();

  constructor(capacity: number) {
    this.capacity = capacity;
  }

  get(key: number): number {
    if (!this.keyTable.has(key)) return -1;

    const node = this.keyTable.get(key)!;
    // жҸҗеҚҮйў‘зҺҮ
    this.increaseFreq(node);
    return node.value;
  }

  put(key: number, value: number): void {
    if (this.capacity === 0) return;

    if (this.keyTable.has(key)) {
      // е·ІеӯҳеңЁпјҢжӣҙж–°еҖје№¶жҸҗеҚҮйў‘зҺҮ
      const node = this.keyTable.get(key)!;
      node.value = value;
      this.increaseFreq(node);
      return;
    }

    // зј“еӯҳж»ЎдәҶпјҢж·ҳжұ°йў‘зҺҮжңҖдҪҺзҡ„
    if (this.keyTable.size >= this.capacity) {
      const minFreqList = this.freqTable.get(this.minFreq)!;
      const victim = minFreqList.removeLast()!;
      this.keyTable.delete(victim.key);
    }

    // ж–°е»әиҠӮзӮ№
    const newNode = new DLinkedNode(key, value);
    this.keyTable.set(key, newNode);

    // еҠ е…Ҙ freq=1 зҡ„й“ҫиЎЁ
    if (!this.freqTable.has(1)) {
      this.freqTable.set(1, new DLinkedList());
    }
    this.freqTable.get(1)!.addFirst(newNode);

    // ж–°иҠӮзӮ№йў‘зҺҮдёҖе®ҡжҳҜ 1
    this.minFreq = 1;
  }

  private increaseFreq(node: DLinkedNode): void {
    const oldFreq = node.freq;
    const newFreq = oldFreq + 1;
    node.freq = newFreq;

    // д»Һж—§йў‘зҺҮй“ҫиЎЁдёӯз§»йҷӨ
    const oldList = this.freqTable.get(oldFreq)!;
    oldList.remove(node);

    // еҰӮжһңж—§й“ҫиЎЁз©әдәҶпјҢдё”еҲҡеҘҪжҳҜжңҖе°Ҹйў‘зҺҮ вҶ’ жӣҙж–° minFreq
    if (oldList.size === 0 && this.minFreq === oldFreq) {
      this.minFreq = newFreq;
    }

    // еҠ е…Ҙж–°йў‘зҺҮй“ҫиЎЁеӨҙйғЁ
    if (!this.freqTable.has(newFreq)) {
      this.freqTable.set(newFreq, new DLinkedList());
    }
    this.freqTable.get(newFreq)!.addFirst(node);
  }
}

е®һйҷ…еә”з”ЁеңәжҷҜ вҖӢ

1. ж•°жҚ®еә“жҹҘиҜўзј“еӯҳ вҖӢ

MySQL зҡ„ Query CacheпјҲиҷҪз„¶ 8.0 е·Іеәҹејғпјүе’ҢеҫҲеӨҡ ORM жЎҶжһ¶зҡ„дәҢзә§зј“еӯҳйғҪз”ЁдәҶ LFU зӯ–з•ҘгҖӮзғӯзӮ№ SQL жҹҘиҜўйў‘зҺҮиҝңй«ҳдәҺеҶ·жҹҘиҜўпјҢз”Ё LFU еҸҜд»Ҙдҝқз•ҷжңҖжңүж•Ҳзҡ„зј“еӯҳгҖӮ

2. CDN еҶ…е®№зј“еӯҳ вҖӢ

CDN иҠӮзӮ№зј“еӯҳзғӯй—Ёи§Ҷйў‘/еӣҫзүҮгҖӮйӮЈдәӣзңҹжӯЈзғӯй—Ёзҡ„еҶ…е®№дёҚеә”иҜҘеӣ дёәгҖҢеҲҡеҘҪжңүдәәи®ҝй—®дәҶдёҖе ҶеҶ·й—ЁеҶ…е®№гҖҚе°ұиў«ж·ҳжұ°гҖӮ

3. DNS зј“еӯҳ вҖӢ

жөҸи§ҲеҷЁе’Ңж“ҚдҪңзі»з»ҹзҡ„ DNS зј“еӯҳгҖӮеёёз”ЁеҹҹеҗҚпјҲgoogle.comпјүзҡ„и§Јжһҗз»“жһңеә”иҜҘй•ҝжңҹдҝқз•ҷпјҢиҖҢдёҚжҳҜиў«еҒ¶е°”и®ҝй—®зҡ„й•ҝе°ҫеҹҹеҗҚжҢӨжҺүгҖӮ

4. ж“ҚдҪңзі»з»ҹйЎөйқўзҪ®жҚў вҖӢ

ж“ҚдҪңзі»з»ҹеңЁзү©зҗҶеҶ…еӯҳдёҚеӨҹж—¶пјҢйңҖиҰҒеҶіе®ҡжҠҠе“ӘдәӣйЎөйқўжҚўеҮәеҲ°зЈҒзӣҳгҖӮLinux еҶ…ж ёе°ұдҪҝз”ЁдәҶзұ»дјј LFU зҡ„зӯ–з•ҘпјҲе…·дҪ“жҳҜ Clock-ProпјҢж··еҗҲдәҶ LRU е’Ң LFU зҡ„жҖқжғіпјүгҖӮ
